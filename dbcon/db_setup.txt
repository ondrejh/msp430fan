Install MySQL:

	#apt-get install mysql-server mysql-client
	set password during the installation (1234 for example)

	git clone https://github.com/PyMySQL/PyMySQL.git
	cd PyMySQL
	#apt-get install python3-setuptools
	#python3 ./setup.py install

Create database:

	run client and input password: #mysql -u root -p

	create database (ebio) and tables (temp, heating), insert rows:
	mysql>
		create database ebio;
		use ebio;
		create table temp (id int, value VARCHAR(16));
		insert into temp (id value) values (1,''),(2,''),(3,''),(4,''),(5,'');
		create table heating (request VARCHAR(32), status VARCHAR(32), fuse VARCHAR(32));
		insert into heating (request,status,fuse) values ('','','');
		create table progstat (request VARCHAR(32), status VARCHAR(32), tstamp TIMESTAMP);
		insert into progstat (request,status) values ('','');
		create table log (t1 VARCHAR(32), t2 VARCHAR(32), t3 VARCHAR(32), t4 VARCHAR(32), t5 VARCHAR(32), h varchar(32), f varchar(32), tstamp timestamp);

	check if it works:
	mysql>
		show databases;
		use ebio;
		show tables;
		select * from temp;
		select * from heating;
		select * from progstat;
